    <script>
        // --- State Management ---
        let state = {
            username: '',
            balance: 0,
            minedTotal: 0,
            miningClicks: 0,
            hashUsed: false,
            history: []
        };

        const miningAmounts = [160000, 600000, 122000, 245000, 324000];
        const correctHash = "HASH#293837382";

        // --- Navigation ---
        function toggleAuth() {
            document.getElementById('register-form').classList.toggle('hidden');
            document.getElementById('login-form').classList.toggle('hidden');
        }

        function showPage(pageName) {
            // Hide all sections
            const sections = ['auth', 'dashboard', 'miner-intro', 'miner-active', 'withdraw', 'profile', 'history'];
            sections.forEach(s => document.getElementById(s + '-section').classList.add('hidden'));

            // Show target
            document.getElementById(pageName + '-section').classList.remove('hidden');

            if(pageName === 'history') renderHistory();
        }

        // --- Auth Logic ---
        function handleRegister() {
            const user = document.getElementById('reg-username').value;
            if(!user) return alert("Please enter a username");
            state.username = user;
            enterDashboard();
        }

        function handleLogin() {
            const user = document.getElementById('login-username').value;
            if(!user) return alert("Please enter a username");
            state.username = user;
            enterDashboard();
        }

        function enterDashboard() {
            document.getElementById('dash-username').innerText = state.username;
            updateBalances();
            showPage('dashboard');
            startFakeNotifications();
        }

        function updateBalances() {
            document.getElementById('total-balance').innerText = `NGN ${state.balance.toLocaleString()}.00`;
            document.getElementById('mined-balance').innerText = `NGN ${state.minedTotal.toLocaleString()}.00`;
        }

        // --- Mining Logic ---
        function validateHash() {
            const input = document.getElementById('hash-input').value.trim();
            if(input === correctHash) {
                // Reset mining session if it's a new valid hash entry
                if(state.miningClicks >= 5) {
                    state.miningClicks = 0; 
                }
                document.getElementById('clicks-left').innerText = 5 - state.miningClicks;
                showPage('miner-active');
            } else {
                alert("Invalid hash code");
            }
        }

        function executeMine() {
            if(state.miningClicks >= 5) {
                alert("Session expired. Please get another hash code.");
                return;
            }

            const amount = miningAmounts[state.miningClicks];
            state.balance += amount;
            state.minedTotal += amount;
            state.miningClicks++;

            // Update UI
            updateBalances();
            document.getElementById('clicks-left').innerText = 5 - state.miningClicks;

            // Feedback
            const status = document.getElementById('mining-status');
            status.innerText = `You just mined NGN ${amount.toLocaleString()}`;
            setTimeout(() => status.innerText = "", 3000);

            // Add to history
            addHistory(`Mined`, amount, 'plus');

            // Check completion
            if(state.miningClicks >= 5) {
                document.getElementById('mine-btn').innerText = "Session Expired";
                document.getElementById('mine-btn').style.background = "#555";
            }
        }

        // --- Withdrawal Logic ---
        function processWithdrawal() {
            const amount = parseFloat(document.getElementById('withdraw-amount').value);
            if(!amount || amount <= 0) return alert("Enter a valid amount");
            if(amount > state.balance) return alert("Insufficient balance");

            state.balance -= amount;
            updateBalances();
            addHistory('Withdrawal', amount, 'minus');

            const msg = `Congratulations your withdrawal NGN${amount.toLocaleString()} has been successful sent to your account, it will arrive your account 1-3 minutes and if not received contact your bank`;
            document.getElementById('popup-message').innerText = msg;
            document.getElementById('success-popup').classList.remove('hidden');
        }

        function closePopup() {
            document.getElementById('success-popup').classList.add('hidden');
            showPage('dashboard');
        }

        // --- Profile Logic ---
        function copyCode() {
            const code = document.getElementById('page-code').innerText;
            navigator.clipboard.writeText(code);
            alert("Code copied!");
        }

        // --- History Logic ---
        function addHistory(type, amount, cssClass) {
            const date = new Date().toLocaleDateString();
            state.history.unshift({ type, amount, cssClass, date });
        }

        function renderHistory() {
            const container = document.getElementById('history-list');
            if(state.history.length === 0) {
                container.innerHTML = '<p class="text-muted text-center">No history yet.</p>';
                return;
            }
            
            let html = '';
            state.history.forEach(item => {
                const sign = item.cssClass === 'plus' ? '+' : '-';
                html += `
                    <div class="history-item">
                        <div>
                            <div style="font-weight:600">${item.type}</div>
                            <div style="font-size:10px; color:#888">${item.date}</div>
                        </div>
                        <div class="hist-amount ${item.cssClass}">
                            ${sign} NGN ${item.amount.toLocaleString()}
                        </div>
                    </div>
                `;
            });
            container.innerHTML = html;
        }

        // --- Fake Notifications ---
        function startFakeNotifications() {
            setInterval(() => {
                const names = ["James", "Sarah", "Emeka", "Chinedu", "Aisha", "Tunde", "Grace", "Emmanuel"];
                const amounts = ["200,000", "600,000", "45,000", "150,000", "1,200,000", "320,000"];
                
                const randomName = names[Math.floor(Math.random() * names.length)];
                const randomAmount = amounts[Math.floor(Math.random() * amounts.length)];
                
                // Obfuscate name
                const maskedName = randomName.substring(0, 2) + "***";

                const toast = document.getElementById('toast-notification');
                toast.innerText = `${maskedName} Just withdrew NGN${randomAmount}`;
                toast.classList.add('show-toast');

                setTimeout(() => {
                    toast.classList.remove('show-toast');
                }, 3000);

            }, 5000); // Every 5 seconds
        }
